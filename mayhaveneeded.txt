  public void BuyBook(string title, int quantity)
        {
            // Check if the book exists in the library
            Book existingBook = Books.FirstOrDefault(book =>
                book.Title.Equals(title, StringComparison.OrdinalIgnoreCase));

            if (existingBook != null && existingBook.Quantity >= quantity)
            {
                // If the book exists and there are enough copies, decrease the quantity
                existingBook.Quantity -= quantity;
                Console.WriteLine($"{quantity} copies of '{title}' bought successfully. Remaining quantity: {existingBook.Quantity}");

                // Save the updated book data to the text file after buying books
                SaveBookData();
            }
            else
            {
                Console.WriteLine($"Not enough copies of '{title}' available to buy.");
            }
        }
